<div class="section">
<h2>Configuration</h2>
<pre>
username = "username"
message_text = "text"
</pre>

<ul>
    <li><code>username</code> → TikTok username (without @)</li>
    <li><code>message_text</code> → Message you want to send</li>
</ul>
</div>
